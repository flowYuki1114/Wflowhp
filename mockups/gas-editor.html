6function createEstimate(formData) {
7 var ss = SpreadsheetApp.getActiveSpreadsheet();
8 var priceSheet = ss.getSheetByName('単価マスタ');
9 var template = ss.getSheetByName('見積テンプレート');
10
11 // 材料費・工賃を単価マスタから自動取得
12 var materials = lookupMaterials_(priceSheet, formData.items);
13 var laborCost = calculateLabor_(formData.area, formData.type);
14
15 // 粗利率を考慮した最終見積額を算出
16 var subtotal = materials.total + laborCost;
17 var margin = subtotal * 0.25; // 粗利 25%
18 var tax = (subtotal + margin) * 0.10;
19
20 // テンプレートに流し込み → PDF化
21 var newSheet = template.copyTo(ss);
22 fillEstimateData_(newSheet, {
23 customer: formData.customerName,
24 project: formData.projectName,
25 materials: materials.breakdown,
26 labor: laborCost,
27 subtotal: subtotal,
28 margin: margin,
29 tax: tax,
30 total: subtotal + margin + tax
31 });
32
33 var pdf = exportSheetAsPdf_(ss.getId(), newSheet.getSheetId());
34
35 // 顧客にメール自動送信
36 GmailApp.sendEmail(formData.email,
37 '【お見積書】' + formData.projectName,
38 'お見積書を添付いたします。ご確認ください。',
39 { attachments: [pdf] }
40 );
41
42 // 顧客DBに記録
43 logToCustomerDb_(formData.customerName, '見積送付', new Date());
44 Logger.log('見積書送信完了: ' + formData.projectName);
45}
